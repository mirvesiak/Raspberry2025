cmake_minimum_required(VERSION 3.16)
project(Raspberry2025)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_FLAGS "-Wall -Wextra -O2")

# Find packages
find_package(OpenCV REQUIRED)
find_package(PkgConfig REQUIRED)
pkg_check_modules(PIGPIO REQUIRED libpigpio)

# Add your source files
add_executable(rpi_robot
    src/main.cpp
    src/motor_controller.cpp
    src/camera_streamer.cpp
)

# Tell CMake where to find headers & link libs
target_include_directories(rpi_robot PRIVATE
    include
    ${PIGPIO_INCLUDE_DIRS}
)
target_link_libraries(rpi_robot
    ${OpenCV_LIBS}
    ${PIGPIO_LIBRARIES}
    uWS  pthread
)
